SET(LIBRARY_OUTPUT_PATH ${EXECUTABLE_OUTPUT_PATH})

FILE( GLOB QWT_SRC *.cpp )
FILE( GLOB QWT_HDRS *.h )

QT4_WRAP_CPP( QWT_MOC ${QWT_HDRS} )

ADD_LIBRARY( qwt
#  STATIC
  SHARED
  ${QWT_SRC}
  ${QWT_MOC}
  )

TARGET_LINK_LIBRARIES( qwt
  qwt
  ${QT_LIBRARIES}
  )

IF ( WIN32 )
  INSTALL(TARGETS qwt DESTINATION bin/../)
ELSE ( WIN32 )
  INSTALL(TARGETS qwt LIBRARY DESTINATION bin/../)
ENDIF( WIN32 )
